# 节流
一定时间内，如果有新事件触发会被无视

```javascript
function throttle(fn, time) {
  let timer = null
  return function () {
    let _this = this
    let arg = arguments
    if (timer) { // 与节流的区别在这里
      return
    }
    timer = setTimeout(function () {
      fn.apply(_this, arg) // 绑定this， 否则this指向window
      timer = null
    }, time || 300)
  }
}
```
